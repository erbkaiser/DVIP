character_event = {
	id = 7200
	
	picture = "event_battle"
	
	trigger = {
		condition = { type = difficulty value = 1 }
		condition = { type = not value = { type = ai } }
		condition = { type = not value = { type = gold value = 0 } }
		condition = { type = atwar }
	}

	mean_time_to_happen = {
		months = 180

		modifier = {
			condition = { type = difficulty value = 2 }
			factor = 0.8
		}

		modifier = {
			condition = { type = difficulty value = 3 }
			factor = 0.8
		}

		modifier = {
			condition = { type = difficulty value = 4 }
			factor = 0.8
		}
		modifier = {
			condition = { type = stability value = 1 }
			factor = 1.1
		}
		modifier = {
			condition = { type = stability value = 2 }
			factor = 1.1
		}
		modifier = {
			condition = { type = stability value = 3 }
			factor = 1.1
		}
		modifier = {
			condition = { type = not value = { type = stability value = 0 } }
			factor = 0.9
		}
		modifier = {
			condition = { type = not value = { type = stability value = -1 } }
			factor = 0.9
		}
		modifier = {
			condition = { type = not value = { type = stability value = -2 } }
			factor = 0.9
		}

	}

	immidiate = {#We cannot sustain our army anymore !
		effect = { type = prestige value = -25 }
		effect = { type = remove_regiment }
		}

}
